# 每周一道算法题：

[27. 移除元素](https://leetcode.cn/problems/remove-element/)

```js
/**
 * @param {number[]} nums
 * @param {number} val
 * @return {number}
 */
var removeElement = function(nums, val) {
    for (let i=nums.length-1;i>=0;i--) {
        if (nums[i] === val) {
            nums[i] = null;
            for (let j=i;j<nums.length-1;j++) {
                nums[j] = nums[j+1];
                nums[j+1] = null;
            }
        }
    }
    for (let i=0;i<nums.length;i++) {
        if (nums[i] === null) {
            return i;
        }
    }
};
```
